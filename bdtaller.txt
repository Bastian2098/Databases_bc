create table DEPARTAMENTO(
	codigo int NOT NULL,
	nombre varchar(30) NOT NULL,
	primary key(codigo)
);

create table PRESUPUESTO(
	codigo int NOT NULL,
	valor int NOT NULL,
	anio varchar(4) NOT NULL,
	fkDepartamento int NOT NULL,
	primary key(codigo)
);

create table SEDE(
	codigo int NOT NULL,
	nombre varchar(30) NOT NULL,
	poblacion int NOT NULL,
	fkEmpleado int NOT NULL,
	primary key(codigo)
);

create table EMPLEADO(
	id int NOT NULL,
	nombre varchar(30) NOT NULL,
	telefono varchar(15) NOT NULL,
	fecIngreso date NOT NULL,
	fkDepartamento int NOT NULL,
	primary key(id)
);

create table HABILIDAD(
	codigo int NOT NULL,
	descripcion varchar(100) NOT NULL,
	primary key(codigo)
);

create table DEP_SED(
	fkDepartamento int NOT NULL,
	fkSede int NOT NULL,
	direccion varchar(15) NOT NULL
);

create table EMP_HAB(
	fkEmpleado int NOT NULL,
	fkHabilidad int NOT NULL,
	salario int NOT NULL
);

create table SED_EMP(
	fkEmpleado int NOT NULL,
	fkSede int NOT NULL
);

alter table PRESUPUESTO

	foreign key(fkDepartamento)
	references DEPARTAMENTO(codigo);

	references DEPARTAMENTO(codigo);
	
alter table DEP_SED
	add constraint fkSede
	foreign key(fkSede)
	references SEDE(codigo);
	
alter table SEDE
	add constraint fkEmpleado
	foreign key(fkEmpleado)
	references EMPLEADO(id);
	
alter table EMPLEADO
	add constraint fkDepartamento
	foreign key(fkDepartamento)
	references DEPARTAMENTO(codigo);
	
alter table EMP_HAB
	add constraint fkEmpleado
	foreign key(fkEmpleado)
	references EMPLEADO(id);
	
alter table EMP_HAB
	add constraint fkHabilidad
	foreign key(fkHabilidad)
	references HABILIDAD(codigo);
	
alter table SED_EMP
	add constraint fkEmpleado
	foreign key(fkEmpleado)
	references EMPLEADO(id);

alter table SED_EMP
	add constraint fkSede
	foreign key(fkSede)
	references SEDE(codigo);
